# 1. Ver en qué rama estoy
git branch
# o también
git status

# 2. Crear y cambiarme a una rama nueva
git checkout -b nombre-de-la-rama

# 3. Ver todas las ramas (locales y remotas)
git branch              # solo locales
git branch -a           # locales y remotas

# 4. Hacer cambios, agregarlos y hacer commit
git add .
git commit -m "Mensaje descriptivo del cambio"

# 5. Subir la rama al repositorio remoto
git push origin nombre-de-la-rama

# 6. Volver a la rama principal (main)
git checkout main

# 7. Traer los últimos cambios de la rama principal desde remoto
git pull

# 8. Fusionar la rama de trabajo en main (una vez terminado el trabajo)
git merge nombre-de-la-rama

# 9. Subir al remoto los cambios fusionados en main
git push origin main

# 10. Borrar la rama localmente (cuando ya no la necesitas)
git branch -d nombre-de-la-rama

# 11. Borrar la rama en el repo remoto
git push origin --delete nombre-de-la-rama
